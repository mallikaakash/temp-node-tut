# callbacks
Ans: Simplest way of implementation and can lead to callback Hell
     
     function fetchData(url, callback) {
  // Create a new XMLHttpRequest object.
  const xhr = new XMLHttpRequest();

  // Set a callback function to be executed when the state of the request changes.
  xhr.onreadystatechange = function() {
    // Check if the request has been completed and the response has been received.
    if (xhr.readyState === 4 && xhr.status === 200) {
      // Call the callback function with the response data.
      callback(xhr.responseText);
    }
  };

  // Open the request and send it to the server.
  xhr.open("GET", url);
  xhr.send();
}

// Call the fetchData function with a URL and a callback function to handle the response.
fetchData("https://example.com", function(response) {
  console.log(response);
});
